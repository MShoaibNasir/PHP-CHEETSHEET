 }

  4. Multiple inheritance example in advance php 

         


    class father{
        public $a;
        public $b;
        function waoo($x,$y){
           $this->a=$x;
           $this->b=$y;
        }
     }
     
     
     class son extends father{
        public $c=10;
        public $sum;
        function sum(){
          $this->sum= $this->a+$this->b+$this->c;
        }
     }
     class grandSon extends son{
     function display(){
        return $this->sum;
     }
     }
     
     $mom=new grandSon;
     $mom->waoo(4,6);
     $mom->sum();
     echo $mom->display();
